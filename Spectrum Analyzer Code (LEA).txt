#include <FastLED.h>
#include "arduinoFFT.h"

// ============================================================
// LED CONFIG
// ============================================================
#define LED_PIN 5
#define NUM_LEDS 256
#define MATRIX_WIDTH 16
#define MATRIX_HEIGHT 16
#define BRIGHTNESS 60
#define LED_TYPE WS2812B
#define COLOR_ORDER GRB

CRGB leds[NUM_LEDS];

// ============================================================
// MIC / FFT CONFIG
// ============================================================
#define MIC_PIN 34
#define SAMPLES 128
#define SAMPLING_FREQ 10000

ArduinoFFT<double> FFT;
double vReal[SAMPLES];
double vImag[SAMPLES];

// ============================================================
// SPECTRUM DISPLAY CONFIG
// ============================================================
const int NUM_BANDS = MATRIX_WIDTH;
int bandValues[16] = {0};
int peakValues[16] = {0};
int peakHold[16]   = {0};

// 16 bands -> 17 edges (avoid bins 0-1 which are DC / near-DC)
const uint8_t bandEdges[17] = {
  2, 3, 4, 5, 6, 7,  9, 11,
  14, 17, 20, 24, 28, 33, 38, 44, 52
};

// ============================================================
// TUNING KNOBS - OPTIMIZED VALUES
// ============================================================

// How tall bars get for a given signal (INCREASED for distant music)
const double SENSITIVITY_GAIN = 4.5;  // Was 2.6, now picks up quiet/distant music

// Smooth bar motion (0.4 = snappy, 0.7 = smooth)
double smoothed[16] = {0};
const double SMOOTHING = 0.70;

// Gate tiny FFT magnitudes (RAISED to filter electrical noise without capacitor)
const double FFT_NOISE_GATE = 20.0;  // Was 6.0, now filters noise better

// AGC prevents constant pegging at the top (IMPROVED for dynamic range)
double levelMax = 650.0;
const double AGC_RISE = 0.25;         // Was 0.18, faster response to loud music
const double AGC_FALL = 0.008;        // Was 0.03, MUCH slower decay for better range
const double LEVELMAX_FLOOR = 120.0;  // Was 320, MUCH lower allows quiet music to show

// Peak dot behavior
const int PEAK_HOLD_FRAMES = 5;
const int PEAK_FALL_RATE = 3;
const int SILENT_PEAK_FALL_RATE = 5;

// ============================================================
// SILENCE DETECTION (adaptive baseline + hysteresis + debounce)
// ============================================================

double ampSmoothed = 0.0;
const double AMP_DECAY = 0.90;

// Baseline tracks "quiet" environment slowly
double noiseBaseline = 0.0;

// Baseline adaptation rates
const double BASELINE_ALPHA_FAST = 0.02;   // warmup
const double BASELINE_ALPHA_SLOW = 0.0015; // steady

unsigned long startMs = 0;
const unsigned long WARMUP_MS = 2500;

// Hysteresis thresholds (derived from baseline each frame)
double enterThreshold = 0.0;
double exitThreshold  = 0.0;

// Enter/Exit multipliers (TUNED for better distance detection)
const double ENTER_MULT = 1.25;  // Was 1.4, easier to trigger from distance
const double ENTER_ADD  = 5.0;   // Was 8.0, lower threshold

const double EXIT_MULT  = 1.5;   // Was 1.6, stays active longer
const double EXIT_ADD   = 5.0;   // Was 8.0, lower threshold

// Debounce: require N consecutive frames before switching
const uint8_t ENTER_FRAMES_REQUIRED = 4;
const uint8_t EXIT_FRAMES_REQUIRED  = 8;

uint8_t enterCount = 0;
uint8_t exitCount  = 0;

bool spectrumMode = false;

// ============================================================
// IDLE GLOW (improved radial animation)
// ============================================================
const uint8_t IDLE_GLOW_MIN = 18;
const uint8_t IDLE_GLOW_MAX = 90;
const uint16_t IDLE_GLOW_PERIOD_MS = 2600;
uint8_t idleHue = 150;

// ============================================================
// 5x7 FONT (used for scroll)
// ============================================================
const byte FONT_G[7] = {0b01110,0b10001,0b10000,0b10111,0b10001,0b10001,0b01110};
const byte FONT_R[7] = {0b11110,0b10001,0b10001,0b11110,0b10100,0b10010,0b10001};
const byte FONT_A[7] = {0b01110,0b10001,0b10001,0b11111,0b10001,0b10001,0b10001};
const byte FONT_C[7] = {0b01110,0b10001,0b10000,0b10000,0b10000,0b10001,0b01110};
const byte FONT_E[7] = {0b11111,0b10000,0b10000,0b11110,0b10000,0b10000,0b11111};
const byte FONT_M[7] = {0b10001,0b11011,0b10101,0b10101,0b10001,0b10001,0b10001};
const byte FONT_L[7] = {0b10000,0b10000,0b10000,0b10000,0b10000,0b10000,0b11111};
const byte FONT_I[7] = {0b11111,0b00100,0b00100,0b00100,0b00100,0b00100,0b11111};

// Message: "LEA MARIE"
const byte* message[] = {FONT_L, FONT_E, FONT_A, NULL, FONT_M, FONT_A, FONT_R, FONT_I, FONT_E};
const int MESSAGE_LENGTH = 9;
const int LETTER_WIDTH = 6;
const int TOTAL_WIDTH = MESSAGE_LENGTH * LETTER_WIDTH;

// ============================================================
// FORWARD DECLS
// ============================================================
void scrollMessage(int loops);
void drawLetter(const byte* letter, int startX, int startY, CRGB color, int height);
void mapFrequenciesToBands(bool silent);
void displaySpectrum();
void displayIdleGlow();
int getLedIndex(int x, int y);

// ============================================================
// SETUP
// ============================================================
void setup() {
  Serial.begin(115200);

  FastLED.addLeds<LED_TYPE, LED_PIN, COLOR_ORDER>(leds, NUM_LEDS);
  FastLED.setBrightness(BRIGHTNESS);
  FastLED.clear();
  FastLED.show();

  pinMode(MIC_PIN, INPUT);

  // ESP32 ADC setup
  analogReadResolution(12);
  analogSetPinAttenuation(MIC_PIN, ADC_11db);

  // scroll name
  scrollMessage(3);

  // start adaptive baseline timing
  startMs = millis();

  // show glow immediately so it never looks dead
  displayIdleGlow();
  FastLED.show();
}

// ============================================================
// LOOP
// ============================================================
void loop() {
  // 1) Sample audio
  unsigned long us = micros();
  for (int i = 0; i < SAMPLES; i++) {
    vReal[i] = analogRead(MIC_PIN);
    vImag[i] = 0;
    while (micros() - us < (1000000UL / SAMPLING_FREQ)) {}
    us += (1000000UL / SAMPLING_FREQ);
  }

  // 2) Remove DC + compute amplitude (for silence detection)
  double mean = 0.0;
  for (int i = 0; i < SAMPLES; i++) mean += vReal[i];
  mean /= SAMPLES;

  double amp = 0.0;
  for (int i = 0; i < SAMPLES; i++) {
    vReal[i] -= mean;
    amp += fabs(vReal[i]);
  }
  amp /= SAMPLES;

  ampSmoothed = ampSmoothed * AMP_DECAY + amp * (1.0 - AMP_DECAY);

  // 3) Baseline + thresholds
  bool warmingUp = (millis() - startMs < WARMUP_MS);
  if (noiseBaseline < 0.1) noiseBaseline = ampSmoothed;

  enterThreshold = noiseBaseline * ENTER_MULT + ENTER_ADD;
  exitThreshold  = noiseBaseline * EXIT_MULT  + EXIT_ADD;

  // safety minimums
  if (enterThreshold < 25) enterThreshold = 25;
  if (exitThreshold  < 18) exitThreshold  = 18;

  // Update baseline ONLY when confidently quiet (below exit threshold),
  // so music doesn't poison the baseline.
  if (warmingUp || ampSmoothed < exitThreshold) {
    double a = warmingUp ? BASELINE_ALPHA_FAST : BASELINE_ALPHA_SLOW;
    noiseBaseline = noiseBaseline * (1.0 - a) + ampSmoothed * a;
  }

  // 4) Mode switching with hysteresis + debounce
  if (!spectrumMode) {
    if (ampSmoothed > enterThreshold) {
      if (++enterCount >= ENTER_FRAMES_REQUIRED) {
        spectrumMode = true;
        enterCount = 0;
        exitCount = 0;
      }
    } else {
      enterCount = 0;
    }
  } else {
    if (ampSmoothed < exitThreshold) {
      if (++exitCount >= EXIT_FRAMES_REQUIRED) {
        spectrumMode = false;
        exitCount = 0;
        enterCount = 0;
      }
    } else {
      exitCount = 0;
    }
  }

  // 5) FFT
  FFT.windowing(vReal, SAMPLES, FFT_WIN_TYP_HAMMING, FFT_FORWARD);
  FFT.compute(vReal, vImag, SAMPLES, FFT_FORWARD);
  FFT.complexToMagnitude(vReal, vImag, SAMPLES);

  // 6) Map + display
  mapFrequenciesToBands(!spectrumMode);

  if (!spectrumMode) displayIdleGlow();
  else displaySpectrum();

  FastLED.show();
  delay(10);
}

// ============================================================
// MAP FREQUENCIES -> BANDS
// ============================================================
void mapFrequenciesToBands(bool silent) {
  if (silent) {
    for (int i = 0; i < NUM_BANDS; i++) {
      smoothed[i] = 0;
      bandValues[i] = 0;
      peakValues[i] = max(0, peakValues[i] - SILENT_PEAK_FALL_RATE);
      peakHold[i] = 0;
    }
    levelMax = max(levelMax, LEVELMAX_FLOOR);
    return;
  }

  double raw[16];
  double peak = 0.0;

  for (int i = 0; i < NUM_BANDS; i++) {
    double sum = 0.0;
    int cnt = 0;
    for (int j = bandEdges[i]; j < bandEdges[i + 1]; j++) {
      sum += vReal[j];
      cnt++;
    }
    double val = (cnt > 0) ? (sum / cnt) : 0.0;

    if (val < FFT_NOISE_GATE) val = 0.0;
    raw[i] = val;

    if (val > peak) peak = val;
  }

  // AGC update
  if (peak > levelMax) levelMax = levelMax + (peak - levelMax) * AGC_RISE;
  else levelMax = max(LEVELMAX_FLOOR, levelMax * (1.0 - AGC_FALL));

  for (int i = 0; i < NUM_BANDS; i++) {
    double normalized = (levelMax > 0) ? (raw[i] / levelMax) : 0.0;
    double h = normalized * MATRIX_HEIGHT * SENSITIVITY_GAIN;
    h = constrain(h, 0, MATRIX_HEIGHT);

    smoothed[i] = smoothed[i] * (1.0 - SMOOTHING) + h * SMOOTHING;
    int out = (int)(smoothed[i] + 0.5);
    bandValues[i] = constrain(out, 0, MATRIX_HEIGHT);

    // peak dot
    if (bandValues[i] > peakValues[i]) {
      peakValues[i] = bandValues[i];
      peakHold[i] = PEAK_HOLD_FRAMES;
    } else if (peakHold[i] > 0) {
      peakHold[i]--;
    } else {
      peakValues[i] = max(0, peakValues[i] - PEAK_FALL_RATE);
    }
  }
}

// ============================================================
// DISPLAY: SPECTRUM
// ============================================================
void displaySpectrum() {
  FastLED.clear();

  for (int x = 0; x < NUM_BANDS; x++) {
    int h = bandValues[x];

    for (int y = 0; y < h; y++) {
      int idx = getLedIndex(x, MATRIX_HEIGHT - 1 - y);
      uint8_t hue = (uint8_t)map(y, 0, MATRIX_HEIGHT, 0, 160);
      leds[idx] = CHSV(hue, 255, 255);
    }

    if (peakValues[x] > 0) {
      int p = constrain(peakValues[x], 0, MATRIX_HEIGHT - 1);
      leds[getLedIndex(x, MATRIX_HEIGHT - 1 - p)] = CRGB::White;
    }
  }
}

// ============================================================
// DISPLAY: IDLE GLOW (smooth radial animation)
// ============================================================
void displayIdleGlow() {
  // "Breathing" brightness using smooth sine wave
  uint16_t phase = (uint16_t)((millis() % IDLE_GLOW_PERIOD_MS) * 65535UL / IDLE_GLOW_PERIOD_MS);
  uint8_t breath = sin8(phase >> 8); // 0..255 smoothly
  uint8_t baseV  = (uint8_t)map(breath, 0, 255, IDLE_GLOW_MIN, IDLE_GLOW_MAX);

  // Slow hue drift
  static uint32_t lastHueStep = 0;
  if (millis() - lastHueStep > 40) {
    idleHue++;
    lastHueStep = millis();
  }

  // Radial falloff around center (removes stripe look)
  const int cx = (MATRIX_WIDTH - 1) / 2;   // 7
  const int cy = (MATRIX_HEIGHT - 1) / 2; // 7
  const uint16_t maxD2 = 98; // Max distance^2 from center to corner

  // Fill every LED with computed glow
  for (int y = 0; y < MATRIX_HEIGHT; y++) {
    for (int x = 0; x < MATRIX_WIDTH; x++) {
      int dx = x - cx;
      int dy = y - cy;
      uint16_t d2 = (uint16_t)(dx * dx + dy * dy);

      // Falloff: center bright, edges dim
      uint8_t fall = (uint8_t)map((int)d2, 0, (int)maxD2, 255, 90);

      // Combine breath with falloff
      uint8_t v = qmul8(baseV, fall);

      // Gentle concentric hue shift based on distance
      uint8_t hue = (uint8_t)(idleHue + (d2 >> 1));

      leds[getLedIndex(x, y)] = CHSV(hue, 170, v);
    }
  }
}

// ============================================================
// SCROLL TEXT
// ============================================================
void scrollMessage(int loops) {
  for (int loop = 0; loop < loops; loop++) {
    for (int offset = MATRIX_WIDTH; offset > -TOTAL_WIDTH; offset--) {
      FastLED.clear();

      int xPos = offset;
      for (int i = 0; i < MESSAGE_LENGTH; i++) {
        if (message[i] != NULL) {
          uint8_t hue = (uint8_t)((i * 30 + loop * 20) % 256);
          drawLetter(message[i], xPos, 4, CHSV(hue, 255, 255), 7);
        }
        xPos += LETTER_WIDTH;
      }

      FastLED.show();
      delay(80);
    }
  }

  FastLED.clear();
  FastLED.show();
  delay(150);
}

void drawLetter(const byte* letter, int startX, int startY, CRGB color, int height) {
  for (int row = 0; row < height; row++) {
    byte rowData = letter[row];
    for (int col = 0; col < 5; col++) {
      if (rowData & (1 << (4 - col))) {
        int x = startX + col;
        int y = startY + row;
        if (x >= 0 && x < MATRIX_WIDTH && y >= 0 && y < MATRIX_HEIGHT) {
          leds[getLedIndex(x, y)] = color;
        }
      }
    }
  }
}

// ============================================================
// MATRIX MAPPING (serpentine columns)
// ============================================================
int getLedIndex(int x, int y) {
  if (x % 2 == 0) return x * MATRIX_HEIGHT + y;
  else return x * MATRIX_HEIGHT + (MATRIX_HEIGHT - 1 - y);
}
